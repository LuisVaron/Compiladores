<Document> ::= 'Form'<FreeText><Form>
<Form> ::= 'Begin'<FormComponents>'End'
<FormComponents> ::= <PanelFeatures><Components>
<PanelFeatures> ::= <PanelTitle><Size><Background>
<PanelTitle> ::= 'Title'':''"'<Text>'"'
<Size> ::= 'Size'':''['<Number>','<Number>']'
<Background> ::= 'Background'':'<Color>
<Color> ::= 'WHITE'|'RED'|'GREEN'|'GRAY'
<Components> ::= <Panel>|<TextBox>|<ComboBox>|<CheckBox>|<Button>
<Components> ::= EPSILON
<Panel> ::= 'Panel'<FreeText>'Begin'<PanelFeatures><Components>'End'
<TextBox> ::= 'TextBox'<FreeText>'Begin'<TextBoxFeatures>'End'<Components>?
<TextBoxFeatures> ::= <TextBoxText><Size><Background>
<Text> ::= 'Text'':''"'<Textws>'"'
<ComboBox> ::= 'ComboBox'<FreeText>'Begin'<ComboBoxFeatures>'End'<Components>?
<ComboBoxFeatures> ::= <List><ComboBoxValues><Size><Background>
<List> ::= 'List'':''['<ListValues>']'
<ComboBoxValues> ::= 'Values'':''['<ValuesValues>']'
<ListValues> ::= <FreeText>','<ListValues>|<FreeText>
<ValuesValues> ::= <Number>','<ListValues>|<Number>
<CheckBox> ::= 'CheckBox'<FreeText>'Begin'<List><'End'<Components>?
<Button> ::= 'Button'<FreeText>'Begin'<ButtonFeatures>'End'<Components>?
<ButtonFeatures> ::= <Text><Background><ButtonOnclick>
<ButtonOnclick> ::= 'Onclick'':'<ButtonAction>
<ButtonAction> ::= 'btnOKOnclick'|'btnCancelOnclick'
<Textws> ::= <Freetext>' '|<Freetext>
